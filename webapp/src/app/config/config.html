  # List of the linters which could be used to analyze your repo
  linters:
    - deadcode
    - dupl
    - goconst
    - gocyclo
    - gofmt
    - goimports
    - golint
    - gotype
    - ineffassign
    - interfacer
    - lll
    - vet
    - vetshadow
  # Directories to skip
  skipdirs:
    - skip-dir
  # Version of GOLANG
  go: 1.6
  # Whether use GOLANG vendoring
  vendor: false
